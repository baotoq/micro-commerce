```csharp
using Vogen;

namespace MicroCommerce.ApiService.Features.{Feature}.Domain.ValueObjects;

[ValueObject<Guid>(conversions: Conversions.EfCoreValueConverter | Conversions.SystemTextJson)]
public partial record struct {TypeName}
{
    public static Validation Validate(Guid value) =>
        value != Guid.Empty
            ? Validation.Ok
            : Validation.Invalid("{TypeName} cannot be empty.");

    public static {TypeName} New() => From(Guid.CreateVersion7());
}
```

**Step 2: Update BaseDbContext.cs** — Replace `StronglyTypedIdConvention` registration with `RegisterAllInVogenEfCoreConverters()`:

```csharp
using MicroCommerce.ApiService.Common.Persistence.Conventions;
using Microsoft.EntityFrameworkCore;

namespace MicroCommerce.ApiService.Common.Persistence;

public abstract class BaseDbContext : DbContext
{
    protected BaseDbContext(DbContextOptions options) : base(options)
    {
    }

    protected override void ConfigureConventions(ModelConfigurationBuilder configurationBuilder)
    {
        base.ConfigureConventions(configurationBuilder);

        configurationBuilder.RegisterAllInVogenEfCoreConverters();
        configurationBuilder.Conventions.Add(_ => new AuditableConvention());
        configurationBuilder.Conventions.Add(_ => new Conventions.ConcurrencyTokenConvention());
        configurationBuilder.Conventions.Add(_ => new SoftDeletableConvention());
    }
}
```

Note: `RegisterAllInVogenEfCoreConverters()` is a Vogen source-generated extension method on `ModelConfigurationBuilder`. It replaces the hand-rolled `StronglyTypedIdConvention`.
