```bash
  "$SCRIPT_DIR/base/web/sealed-secret.yaml" \
  --from-literal=auth-secret=k8s-dev-auth-secret-change-in-production \
  --from-literal=keycloak-client-secret=nextjs-app-secret
```


```yaml

      containers:
```
